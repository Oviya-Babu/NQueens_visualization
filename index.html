function solveNQueens(row) {
  if (row === N) return true; // ✓ solved!
  
  for (col = 0; col < N; col++) {
    if (isSafe(board, row, col)) {
      board[row] = col;       // place queen
      if (solveNQueens(row + 1))
        return true;
      board[row] = -1;        // backtrack
    }
  }
  return false; // trigger backtrack
}
